def find_cycle(r, x0=0.2, N=10000, k=256, eps=1e-6):
    """
          m
      x0:   x
      N:   x_n   n \ge N,      x0
      k:     N
      eps:  10^{-6}
       :  (m, u) -     
    """
    #   x_n  N
    x = x0
    for _ in range(N):
        x = logistic_map(x, r)
    
		#     x_n,      ""  (,.   (diff < eps)  )
    uniques = []
    for _ in range(k):
        x = logistic_map(x, r)

        flag = False
        for i, u in enumerate(uniques):
            if abs(u - x) < eps:
                #      diff < eps      u  x
                uniques[i] = 0.5 * (u + x)
                flag = True
                break
        if not flag:
            uniques.append(x)
    m = len(uniques)
    return m, uniques

def lameray(r, x0=0.2, N=50, curve_dots=400):
    """
           
       r    x0.
    curve_dots:   
    """

    xs = np.linspace(0, 1, curve_dots)
    ys = logistic_map(xs, r)

    fig, ax = plt.subplots(figsize=(6, 6))

    ax.plot(xs, ys, label=r"$x_{n+1} = r x_n (1 - x_n)$")
    ax.plot(xs, xs, linestyle="--", label=r"$x_{n+1} = x_n$")

		#  
    x = x0
    y = 0.0 

    for _ in range(N):
        #   
        y_new = logistic_map(x, r)
        ax.plot([x, x], [y, y_new])

        #   
        ax.plot([x, y_new], [y_new, y_new])  

        # 
        x, y = y_new, y_new
       
    #  
    ax.set_xlim(0, 1)
    ax.set_ylim(0, 1)
    ax.set_xlabel(r"$x_n$")
    ax.set_ylabel(r"$x_{n+1}$")
    ax.set_title(fr"$r = {r}, x_0 = {x0}$")
    ax.legend()
    ax.grid(True)

    plt.show()
