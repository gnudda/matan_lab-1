def sequence_iterate(x0, r, amount):
    """
      numpy      {x_n}
    x0:   
    r: 
    amount: - 
     : numpy.array
    """
    x_list = [x0]
    x_cur = x0
    for _ in range(amount):
        x_cur = logistic_map(x_cur, r)
        x_list.append(x_cur)
    return np.array(x_list)
